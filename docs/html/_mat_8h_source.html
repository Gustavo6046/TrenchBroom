360 template <typename T, size_t S>
361 struct MatrixDeterminant {
362  T operator() (const Mat<T,S,S>& mat) const {
363  // Laplace after first col
364  T det = static_cast<T>(0.0);
365  for (size_t r = 0; r < S; r++) {
366  const T f = static_cast<T>(r % 2 == 0 ? 1.0 : -1.0);
367  det += f * mat[0][r] * MatrixDeterminant<T,S-1>()(minorMatrix(mat, r, 0));
368  }
369  return det;
370  }
371 };

680 Mat<T,4,4> planeProjectionMatrix(const T distance, const Vec<T,3>& normal, const Vec<T,3>& direction) {
681  // create some coordinate system where the X and Y axes are contained within the plane
682  // and the Z axis is the projection direction
683  Vec<T,3> xAxis;
684 
685  switch (normal.firstComponent()) {
686  case Math::Axis::AX:
687  xAxis = crossed(normal, Vec<T,3>::PosZ).normalized();
688  break;
689  default:
690  xAxis = crossed(normal, Vec<T,3>::PosX).normalized();
691  break;

694  const Vec<T,3>& zAxis = direction;

697  assert(Math::eq(yAxis.length(), 1.0));
698  assert(Math::eq(zAxis.length(), 1.0));
699 
700  return coordinateSystemMatrix(xAxis, yAxis, zAxis, distance * normal);
