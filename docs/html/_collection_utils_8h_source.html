462  template <typename T, typename Compare>
463  void orderedDifference(std::vector<T>& minuend, const std::vector<T>& subtrahend, const Compare& cmp) {
464  typedef std::vector<T> Vec;
465 
466  typename Vec::iterator mIt = minuend.begin();
467  typename Vec::const_iterator sIt = subtrahend.begin();
468  typename Vec::const_iterator sEnd = subtrahend.end();
469 
470  while (mIt != minuend.end() && sIt != sEnd) {
471  const T& m = *mIt;
472  const T& s = *sIt;
473  if (cmp(m, s)) { // m < s
474  ++mIt;
475  } else if (cmp(s, m)) { // s < m
476  ++sIt;
477  } else { // s == m
478  mIt = minuend.erase(mIt);
479  }
480  }

618  template <typename T, typename Compare>
619  std::vector<T> setUnion(const std::vector<T>& vec1, const std::vector<T>& vec2) {
620  if (vec1.empty())
621  return vec2;
622  if (vec2.empty())
623  return vec1;
624 
625  typedef typename std::vector<T> Vec;
626  typedef typename Vec::const_iterator I;
627 
628  I cur1 = vec1.begin();
629  const I end1 = vec1.end();
630  I cur2 = vec2.begin();
631  const I end2 = vec2.end();
632 
633  Vec result;
634  Compare cmp;
635 
636  while (cur1 != end1 && cur2 != end2) {
637  if (cmp(*cur1, *cur2)) {
638  result.push_back(*cur1);
639  ++cur1;
640  } else if (cmp(*cur2, *cur1)) {
641  result.push_back(*cur2);
642  ++cur2;
643  } else {
644  // both values are equal
645  result.push_back(*cur1);
646  ++cur1;
647  ++cur2;
648  }
649  }
650 
651  while (cur1 != end1)
652  result.push_back(*cur1++);
653  while (cur2 != end2)
654  result.push_back(*cur2++);

803  }
