175  bool TrenchBroomApp::openDocument(const String& pathStr) {
176  MapFrame* frame = NULL;
177  const IO::Path path(pathStr);
178  try {
179  String gameName = "";
180  Model::MapFormat::Type mapFormat = Model::MapFormat::Unknown;
181 
182  const Model::GameFactory& gameFactory = Model::GameFactory::instance();
183  const std::pair<String, Model::MapFormat::Type> detected = gameFactory.detectGame(path);
184  gameName = detected.first;
185  mapFormat = detected.second;
186 
187  if (gameName.empty() || mapFormat == Model::MapFormat::Unknown) {
188  if (!GameDialog::showOpenDocumentDialog(NULL, gameName, mapFormat))
189  return false;
190  }
191 
192  Model::GamePtr game = gameFactory.createGame(gameName);
193  assert(game != NULL);
194 
195  frame = m_frameManager->newFrame();
196  frame->openDocument(game, mapFormat, path);
197  return true;
198  } catch (const FileNotFoundException& e) {
199  m_recentDocuments->removePath(IO::Path(path));
200  if (frame != NULL)
201  frame->Close();
202  ::wxMessageBox(e.what(), "TrenchBroom", wxOK, NULL);
203  return false;
204  } catch (const Exception& e) {
205  if (frame != NULL)
206  frame->Close();
207  ::wxMessageBox(e.what(), "TrenchBroom", wxOK, NULL);
208  return false;
209  } catch (...) {
210  if (frame != NULL)
211  frame->Close();
212  ::wxMessageBox(pathStr + " could not be opened.", "TrenchBroom", wxOK, NULL);
213  return false;
214  }
215  }

305  int TrenchBroomApp::FilterEvent(wxEvent& event) {
306  if (event.GetEventObject() != NULL) {
307  if (event.GetEventType() == wxEVT_ACTIVATE) {
308  m_lastActivation = wxGetLocalTimeMillis();
309  } else if (event.GetEventType() == wxEVT_LEFT_DOWN ||
310  event.GetEventType() == wxEVT_MIDDLE_DOWN ||
311  event.GetEventType() == wxEVT_RIGHT_DOWN ||
312  event.GetEventType() == wxEVT_LEFT_UP ||
313  event.GetEventType() == wxEVT_MIDDLE_UP ||
314  event.GetEventType() == wxEVT_RIGHT_UP) {
315  if (wxGetLocalTimeMillis() - m_lastActivation <= 10)
316  return 1;
317  }
318  }
319  return wxApp::FilterEvent(event);
320  }

328  void TrenchBroomApp::OnUpdateUI(wxUpdateUIEvent& event) {
329  switch (event.GetId()) {
330  case wxID_PREFERENCES:
331  case wxID_ABOUT:
332  case wxID_NEW:
333  case wxID_OPEN:
334  case wxID_EXIT:
335  case wxID_HELP:
336  case CommandIds::Menu::FileOpenRecent:
337  event.Enable(true);
338  break;
339  default:
340  if (event.GetId() >= CommandIds::Menu::FileRecentDocuments &&
341  event.GetId() < CommandIds::Menu::FileRecentDocuments + 10)
342  event.Enable(true);
343  else if (m_frameManager->allFramesClosed())
344  event.Enable(false);
345  break;
346  }
347  }
