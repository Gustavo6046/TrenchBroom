79  const T intersectWithSphere(const Vec<T,S>& position, const T radius) const {
80  const Vec<T,S> diff = origin - position;
81 
82  const T p = static_cast<T>(2.0) * diff.dot(direction);
83  const T q = diff.squaredLength() - radius * radius;
84 
85  const T d = p * p - static_cast<T>(4.0) * q;
86  if (d < static_cast<T>(0.0))
87  return Math::nan<T>();
88 
89  const T s = std::sqrt(d);
90  const T t0 = (-p + s) / static_cast<T>(2.0);
91  const T t1 = (-p - s) / static_cast<T>(2.0);
92 
93  if (t0 < static_cast<T>(0.0) && t1 < static_cast<T>(0.0))
94  return Math::nan<T>();
95  if (t0 > static_cast<T>(0.0) && t1 > static_cast<T>(0.0))
96  return std::min(t0, t1);
97  return std::max(t0, t1);
98  }

100  const T intersectWithSphere(const Vec<T,S>& position, const T radius, const T maxDistance) const {
101  const T distanceToCenter = (position - origin).squaredLength();
102  if (distanceToCenter > maxDistance * maxDistance)
103  return Math::nan<T>();
104 
105  return intersectWithSphere(position, radius * distanceToCenter);

164  const LineDistance squaredDistanceToSegment(const Vec<T,S>& start, const Vec<T,3>& end) const {
165  Vec<T,S> u, v, w;
166  u = end - start;
167  v = direction;
168  w = start - origin;
169 
170  const T a = u.dot(u);
171  const T b = u.dot(v);
172  const T c = v.dot(v);
173  const T d = u.dot(w);
174  const T e = v.dot(w);
175  const T D = a * c - b * b;
176  T sN, sD = D;
177  T tN, tD = D;
178 
179  if (Math::zero(D))
180  return LineDistance::Parallel(w.squaredLength());
181 
182  sN = (b * e - c * d);
183  tN = (a * e - b * d);
184  if (sN < static_cast<T>(0.0)) {
185  sN = static_cast<T>(0.0);
186  tN = e;
187  tD = c;
188  } else if (sN > sD) {
189  sN = sD;
190  tN = e + b;
191  tD = c;
192  }
193 
194  const T sc = Math::zero(sN) ? static_cast<T>(0.0) : sN / sD;
195  const T tc = std::max(Math::zero(tN) ? static_cast<T>(0.0) : tN / tD, static_cast<T>(0.0));
196 
197  u = u * sc; // vector from segment start to the closest point on the segment
198  v = v * tc; // vector from ray origin to closest point on the ray
199  w = w + u;
200  const Vec<T,S> dP = w - v;
201 
202  return LineDistance::NonParallel(tc, dP.squaredLength(), sc);
203  }

211  const LineDistance squaredDistanceToRay(const Ray<T,3>& ray) const {
212  Vec<T,S> u, v, w;
213  u = ray.direction;
214  v = direction;
215  w = ray.origin - origin;
216 
217  const T a = u.dot(u); // other.direction.dot(other.direction) (squared length)
218  const T b = u.dot(v); // other.direction.dot(this.direction)
219  const T c = v.dot(v); // this.direction.dot(this.direction) (squared length)
220  const T d = u.dot(w); // other.direction.dot(origin delta)
221  const T e = v.dot(w); // this.direction.dot(origin delta)
222  const T D = a * c - b * b;
223  T sN, sD = D;
224  T tN, tD = D;
225 
226  if (Math::zero(D))
227  return LineDistance::Parallel(w.squaredLength());
228 
229  sN = (b * e - c * d);
230  tN = (a * e - b * d);
231  if (sN < static_cast<T>(0.0)) {
232  sN = static_cast<T>(0.0);
233  tN = e;
234  tD = c;
235  }
236 
237  const T sc = Math::zero(sN) ? static_cast<T>(0.0) : sN / sD;
238  const T tc = std::max(Math::zero(tN) ? static_cast<T>(0.0) : tN / tD, static_cast<T>(0.0));
239 
240  u = u * sc; // vector from the given ray's origin to the closest point on given ray
241  v = v * tc; // vector from this ray's origin to closest point on this ray

243  const Vec<T,S> dP = w - v;

248  const LineDistance squaredDistanceToLine(const Vec<T,S>& lineAnchor, const Vec<T,S>& lineDir) const {
249  const Vec<T,S> w0 = origin - lineAnchor;
250  const T a = direction.dot(direction);
251  const T b = direction.dot(lineDir);
252  const T c = lineDir.dot(lineDir);
253  const T d = direction.dot(w0);
254  const T e = lineDir.dot(w0);
255 
256  const T f = a * c - b * b;
257  if (Math::zero(f))
258  return LineDistance::Parallel(w0.squaredLength());
259 
260  const T sc = std::max((b * e - c * d) / f, static_cast<T>(0.0));
261  const T tc = (a * e - b * d) / f;
262 
263  const Vec<T,S> rp = origin + sc * direction; // point on ray
264  const Vec<T,S> lp = lineAnchor + tc * lineDir; // point on line
265  return LineDistance::NonParallel(sc, (rp - lp).squaredLength(), tc);
266  }

276 const TT intersectRayWithTriangle(const Ray<TT, 3>& R, const Vec<TT,3>& V0, const Vec<TT,3>& V1, const Vec<TT,3>& V2) {
277  // see http://www.cs.virginia.edu/~gfx/Courses/2003/ImageSynthesis/papers/Acceleration/Fast%20MinimumStorage%20RayTriangle%20Intersection.pdf
278 
279  const Vec<TT,3>& O = R.origin;
280  const Vec<TT,3>& D = R.direction;
281  const Vec<TT,3> E1 = V1 - V0;
282  const Vec<TT,3> E2 = V2 - V0;
283  const Vec<TT,3> P = crossed(D, E2);
284  const TT a = P.dot(E1);
285  if (Math::zero(a))
286  return Math::nan<TT>();
287 
288  const Vec<TT,3> T = O - V0;
289  const Vec<TT,3> Q = crossed(T, E1);
290 
291  const TT t = Q.dot(E2) / a;
292  if (Math::neg(t))
293  return Math::nan<TT>();
294 
295  const TT u = P.dot(T) / a;
296  if (Math::neg(u))
297  return Math::nan<TT>();
298 
299  const TT v = Q.dot(D) / a;
300  if (Math::neg(v))
301  return Math::nan<TT>();
302 
303  if (Math::gt(u+v, static_cast<TT>(1.0)))
304  return Math::nan<TT>();
305 
306  return t;
307 }
