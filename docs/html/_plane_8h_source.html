206  Vec<T,S> project(const Vec<T,S>& point, const Vec<T,S>& direction) const {
207  const T f = direction.dot(normal);
208  if (Math::zero(f))

234 template <typename T>
235 bool setPlanePoints(Plane<T,3>& plane, const Vec<T,3>& point0, const Vec<T,3>& point1, const Vec<T,3>& point2) {
236  const Vec<T,3> v1 = point2 - point0;
237  const Vec<T,3> v2 = point1 - point0;
238  plane.normal = crossed(v1, v2);
239 
240  // Fail if v1 and v2 are parallel, opposite, or either is zero-length.
241  // Rearranging "A cross B = ||A|| * ||B|| * sin(theta) * n" (n is a unit vector perpendicular to A and B) gives sin_theta below
242  const T sin_theta = Math::abs(plane.normal.length() / (v1.length() * v2.length()));
243  if (sin_theta < Math::Constants<T>::angleEpsilon()
244  || Math::isnan(sin_theta)
245  || sin_theta == std::numeric_limits<T>::infinity()
246  || sin_theta == -std::numeric_limits<T>::infinity())
247  return false;
248 
249  plane.normal.normalize();
250  plane.distance = point0.dot(plane.normal);
251  return true;
252 }
