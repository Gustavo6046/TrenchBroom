135 typename Polyhedron<T,FP,VP>::MoveVerticesResult Polyhedron<T,FP,VP>::doMoveVertices(typename V::List positions, const V& delta, const bool allowMergeIncidentVertices, Callback& callback) {
136  std::sort(positions.begin(), positions.end(), typename V::InverseDotOrder(delta));
137  MoveVerticesResult totalResult;

139  for (size_t i = 0; i < positions.size(); ++i) {

180 template <typename T, typename FP, typename VP>
181 typename Polyhedron<T,FP,VP>::SplitResult Polyhedron<T,FP,VP>::splitFace(const typename V::List& vertexPositions, Callback& callback) {
182  Face* face = findFaceByPositions(vertexPositions);
183  if (face == NULL)
184  return SplitResult(false);
185 
186  Vertex* newVertex = new Vertex(face->center());
187  m_vertices.append(newVertex, 1);
188 
189  const size_t vertexCount = face->vertexCount();
190  HalfEdge* current = face->boundary().front();
191 
192  // First, create a new triangle that cuts into the face. The face will be convex until the following loop finishes.
193  {
194  HalfEdge* next = current->next();
195 
196  HalfEdge* fromCenter = new HalfEdge(newVertex);
197  HalfEdge* fromCenterTwin = new HalfEdge(current->origin());
198  HalfEdge* toCenter = new HalfEdge(current->destination());
199  HalfEdge* toCenterTwin = new HalfEdge(newVertex);
200 
201  HalfEdgeList boundaryReplacement;
202  boundaryReplacement.append(fromCenterTwin, 1);
203  boundaryReplacement.append(toCenterTwin, 1);
204  face->replaceBoundary(current, current, fromCenterTwin);
205 
206  HalfEdgeList newFaceBoundary;
207  newFaceBoundary.append(fromCenter, 1);
208  newFaceBoundary.append(current, 1);
209  newFaceBoundary.append(toCenter, 1);
210 
211  Face* newFace = new Face(newFaceBoundary);
212  m_faces.append(newFace, 1);
213  m_edges.append(new Edge(fromCenter, fromCenterTwin), 1);
214  m_edges.append(new Edge(toCenter, toCenterTwin), 1);
215 
216  callback.faceWasSplit(face, newFace);
217  current = next;
218  }
219 
220  // Now just chop off more triangles until only triangles remain.
221  for (size_t i = 0; i < vertexCount - 2; ++i) {
222  HalfEdge* next = current->next();
223  chopFace(face, current, callback);
224  current = next;
225  }
226 
227  return SplitResult(true, newVertex->position());
228 }

298 typename Polyhedron<T,FP,VP>::MoveVertexResult Polyhedron<T,FP,VP>::movePolygonVertex(Vertex* vertex, const V& destination, const bool allowMergeIncidentVertices) {
299  const V originalPosition(vertex->position());
300  Face* face = *m_faces.begin();
301  if (face->pointStatus(destination) != Math::PointStatus::PSInside)
302  return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
303 
304  Vertex* occupant = findVertexByPosition(destination);
305  if (occupant != NULL && occupant != vertex) {
306  HalfEdge* connectingEdge = vertex->findConnectingEdge(occupant);
307  if (connectingEdge == NULL)
308  connectingEdge = occupant->findConnectingEdge(vertex);
309  if (!allowMergeIncidentVertices || connectingEdge == NULL)
310  return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
311 
312  Vertex* origin = connectingEdge->origin();
313  Vertex* destVertex = connectingEdge->destination();
314 
315  Edge* edge = connectingEdge->edge();
316  face->removeFromBoundary(connectingEdge);
317 
318  m_edges.remove(edge);
319  delete edge;
320  delete connectingEdge;
321  m_vertices.remove(origin);

323 
324  return MoveVertexResult(MoveVertexResult::Type_VertexMoved, originalPosition, destVertex);

330 template <typename T, typename FP, typename VP>
331 typename Polyhedron<T,FP,VP>::MoveVertexResult Polyhedron<T,FP,VP>::movePolyhedronVertex(Vertex* vertex, const V& destination, const bool allowMergeIncidentVertices, Callback& callback) {
332  // The idea of this algorithm can be summarized as follows:
333  // First, break up all faces incident to the given vertex so that they become triangles.
334  // Second, examine the line along which the given vertex will travel when moved to the
335  // given location. Determine the closest point on this line where one of the incident triangles
336  // becomes coplanar with one (or more) of its neighbours. This point is represented by the value of
337  // curFrac, which is between 0.0 and 1.0, 0.0 being the original vertex position and 1.0 the destination.
338  // Third, determine whether the vertex will come to rest on another vertex. If that is the case, and
339  // allowMergeIncidentVertices is true, then merge those vertices if possible and conclude the operation.
340  // Otherwise, proceed with the fourth step: Move the vertex to the point we just computed, and merge the
341  // incident faces with their neighbours.
342  // Continue this until the vertex is either deleted during the merging phase, or until the vertex arrives
343  // at its destination position.
344 
345  T lastFrac = 0.0;
346  const V originalPosition = vertex->position();
347  while (!vertex->position().equals(destination, 0.0)) {
348  splitIncidentFaces(vertex, destination, callback);
349 
350  const T curFrac = computeNextMergePoint(vertex, originalPosition, destination, lastFrac);
351  // if (curFrac < 0.0)
352  // return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
353 
354  if (curFrac < 0.0) {
355  // If no merge point could be computed, we might still get away with merging two vertices. Let's try.
356  Vertex* occupant = findVertexByPosition(destination, vertex);
357  HalfEdge* connectingEdge = NULL;
358  if (!allowMergeIncidentVertices || occupant == NULL || (connectingEdge = vertex->findConnectingEdge(occupant)) == NULL) {
359  mergeIncidentFaces(vertex, callback);
360  if (!vertex->position().equals(originalPosition, 0.0)) {
361  const MoveVertexResult result = moveVertex(vertex, originalPosition, false, callback);
362  unused(result);
363  assert(result.moved());
364  }
365  return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
366  }
367 
368  mergeVertices(connectingEdge, callback);
369  assert(!hasVertex(occupant));
370  assert(hasVertex(vertex));
371  vertex->setPosition(destination);
372  return MoveVertexResult(MoveVertexResult::Type_VertexMoved, originalPosition, vertex);
373  }
374 
375  assert(curFrac > lastFrac);
376  lastFrac = curFrac;
377 
378  const V newPosition = originalPosition + lastFrac * (destination - originalPosition);
379  if (denaturedPolyhedron(vertex, newPosition))
380  return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
381 
382  Vertex* occupant = findVertexByPosition(newPosition, vertex);
383  if (occupant != NULL) {
384  HalfEdge* connectingEdge = vertex->findConnectingEdge(occupant);
385  if (!allowMergeIncidentVertices || connectingEdge == NULL) {
386  vertex->setPosition(newPosition);
387  mergeIncidentFaces(vertex, callback);
388  const MoveVertexResult result = moveVertex(vertex, originalPosition, false, callback);
389  unused(result);
390  assert(result.moved());
391  return MoveVertexResult(MoveVertexResult::Type_VertexUnchanged, originalPosition, vertex);
392  }
393  mergeVertices(connectingEdge, callback);
394  assert(!hasVertex(occupant));
395  assert(hasVertex(vertex));
396  }
397 
398  vertex->setPosition(newPosition);
399 
400  const CleanupResult cleanupResult = cleanupAfterVertexMove(vertex, callback);
401  if (cleanupResult.vertexDeleted()) {
402  if (cleanupResult.vertexDeletedByFaceMerge()) {
403  callback.faceDidChange(cleanupResult.containingFace);
404  } else if (cleanupResult.vertexDeletedByEdgeMerge()) {
405  callback.faceDidChange(cleanupResult.containingEdge->firstFace());
406  callback.faceDidChange(cleanupResult.containingEdge->secondFace());
407  }
408  return MoveVertexResult(MoveVertexResult::Type_VertexDeleted, originalPosition);
409  } else {
410  incidentFacesDidChange(vertex, callback);
411  }
412  }
413 
414  return MoveVertexResult(MoveVertexResult::Type_VertexMoved, originalPosition, vertex);
415 }

417 // Splits the faces incident to the given vertex so that the polyhedron cannot become
418 // convex if the vertex is moved to the given destination. Depending on the orientation of
419 // an incident face and the vector from the given vertex's position to the destination,
420 // this includes chopping off triangles of some incident faces, and splitting other faces
421 // into triangle fans.
422 template <typename T, typename FP, typename VP>
423 void Polyhedron<T,FP,VP>::splitIncidentFaces(Vertex* vertex, const V& destination, Callback& callback) {
424  HalfEdge* firstEdge = vertex->leaving();
425  HalfEdge* curEdge = firstEdge;
426 
427  do {
428  HalfEdge* nextEdge = curEdge->nextIncident();
429  Face* face = curEdge->face();
430  if (face->vertexCount() > 3) {
431  const Math::PointStatus::Type status = face->pointStatus(destination);
432  if (status == Math::PointStatus::PSBelow)
433  chopFace(face, curEdge, callback);
434  else
435  splitFace(face, curEdge, callback);
436  }
437  curEdge = nextEdge;
438  } while (curEdge != firstEdge);
439 }

441 // Creates a new face by chopping off one triangle of the given face. The triangle
442 // will have the destination of the given edge, the origin of the given edge, and
443 // the origin of the given edge's predecessor as its vertices.
444 template <typename T, typename FP, typename VP>
445 void Polyhedron<T,FP,VP>::chopFace(Face* face, HalfEdge* halfEdge, Callback& callback) {
446  assert(face->vertexCount() > 3);
447 
448  HalfEdge* next = halfEdge;
449  HalfEdge* previous = next->previous();
450 
451  HalfEdge* newEdge1 = new HalfEdge(previous->origin());
452  HalfEdge* newEdge2 = new HalfEdge(next->destination());
453 
454  face->replaceBoundary(previous, next, newEdge1);
455 
456  assert(next->next() == previous);
457  assert(previous->previous() == next);
458 
459  HalfEdgeList boundary;
460  boundary.append(previous, 2);
461  boundary.append(newEdge2, 1);
462 
463  Face* newFace = new Face(boundary);
464  callback.faceWasSplit(face, newFace);
465 
466  m_faces.append(newFace, 1);
467  m_edges.append(new Edge(newEdge1, newEdge2), 1);
468 }

490 template <typename T, typename FP, typename VP>
491 T Polyhedron<T,FP,VP>::computeNextMergePoint(Vertex* vertex, const V& origin, const V& destination, T lastFrac) const {
492  HalfEdge* firstEdge = vertex->leaving();
493  HalfEdge* curEdge = firstEdge;
494 
495  T minFrac = 1.0;
496  do {
497  minFrac = std::min(minFrac, computeNextMergePointForIncidentNeighbour(curEdge, origin, destination, lastFrac));
498  minFrac = std::min(minFrac, computeNextMergePointForOppositeNeighbour(curEdge, origin, destination, lastFrac));
499  curEdge = curEdge->nextIncident();
500  } while (curEdge != firstEdge);
501  return minFrac;
502 }

504 template <typename T, typename FP, typename VP>
505 T Polyhedron<T,FP,VP>::computeNextMergePointForIncidentNeighbour(HalfEdge* edge, const V& origin, const V& destination, T lastFrac) const {
506  /*
507  We consider the plane made up by the points p1, p2 and p3 of side and next. If the movement
508  of the vertex were to go through this plane, the brush would become convex, which we must prevent.
509 
510  v----p3
511  |\ n |
512  | \ |
513  | \ |
514  | s \|
515  p1---p2
516 
517  */
518 
519  HalfEdge* nextEdge = edge->nextIncident();
520  HalfEdge* nextNextEdge = nextEdge->nextIncident();
521 
522  assert(edge->face()->vertexCount() == 3);
523  assert(nextEdge->face()->vertexCount() == 3);
524 
525  Vertex* v1 = edge->destination();
526  Vertex* v2 = nextEdge->destination();
527  Vertex* v3 = nextNextEdge->destination();
528 
529  const V& p1 = v1->position();
530  const V& p2 = v2->position();
531  const V& p3 = v3->position();
532 
533  Plane<T,3> plane;
534  if (!setPlanePoints(plane, p1, p2, p3)) {
535  // The points are colinear and we cannot determine the move distance - this is an error!
536  return -1.0;
537  }
538 
539  return computeNextMergePointForPlane(origin, destination, plane, lastFrac);
540 }

542 template <typename T, typename FP, typename VP>
543 T Polyhedron<T,FP,VP>::computeNextMergePointForOppositeNeighbour(HalfEdge* edge, const V& origin, const V& destination, T lastFrac) const {
544  /*
545  We consider the boundary plane of the one neighbour to the side which is not incident to the
546  moved vertex. This neighbour is not necessarily a triangle, but that does not matter.
547 
548  p3------
549  / n |
550  / e |
551  v----p2 i |
552  |\ s | g |
553  | \ | h |
554  | \ | b |
555  | \| o |
556  -----p1 u |
557  \ r |
558  ---------
559  */
560 
561  HalfEdge* myBorder = edge->next();
562  HalfEdge* theirBorder = myBorder->twin();
563 
564  Vertex* v1 = edge->destination();
565  Vertex* v2 = theirBorder->origin();
566  Vertex* v3 = theirBorder->previous()->origin();
567 
568  const V& p1 = v1->position();
569  const V& p2 = v2->position();
570  const V& p3 = v3->position();
571 
572  Plane<T,3> plane;
573  if (!setPlanePoints(plane, p1, p2, p3)) {
574  // The points are colinear and we cannot determine the move distance - this is an error!
575  return -1.0;
576  }
577 
578  return computeNextMergePointForPlane(origin, destination, plane, lastFrac);
579 }

581 template <typename T, typename FP, typename VP>
582 T Polyhedron<T,FP,VP>::computeNextMergePointForPlane(const V& origin, const V& destination, const Plane<T,3>& plane, T lastFrac) const {
583  const T origDot = origin.dot(plane.normal) - plane.distance;
584  const T destDot = destination.dot(plane.normal) - plane.distance;
585 
586  // We only consider this a merge point if the origin does not lie (almost) within the plane.
587  if (std::abs(origDot) >= 0.001) {
588  // Does the vertex travel through the plane?
589  if ((origDot > 0.0) != (destDot > 0.0)) {
590  // The distance must grow because we use the original vertex origin.
591  const T frac = std::abs(origDot) / (std::abs(origDot) + std::abs(destDot));
592  if (frac > lastFrac)
593  return frac;
594  }
595  }
596  return 1.0;
597 }

601 template <typename T, typename FP, typename VP>
602 bool Polyhedron<T,FP,VP>::denaturedPolyhedron(const Vertex* vertex, const V& newPosition) const {
603  const Vertex* firstVertex = m_vertices.front();

632 // Merges the origin and destination vertex of the given edge into one vertex, thereby
633 // deleting the edge, the destination vertex, and the faces incident to the connecting edge
634 // and its twin.
635 // Assumes that these incident faces are triangles.
636 template <typename T, typename FP, typename VP>
637 void Polyhedron<T,FP,VP>::mergeVertices(HalfEdge* connectingEdge, Callback& callback) {
638  HalfEdge* opposingEdge = connectingEdge->twin();
639 
640  Vertex* origin = connectingEdge->origin();
641  Vertex* destination = opposingEdge->origin();
642 
643  // First we merge the triangles that will become invalid by the merge to their neighbours.
644  // We assume they are both triangles.
645  assert(connectingEdge->face()->vertexCount() == 3);
646  assert(opposingEdge->face()->vertexCount() == 3);
647 
648  // We choose two edges of the faces incident to the connecting edge so that these edges do
649  // not share a face! Otherwise the destination vertex would already get deleted during
650  // the neighbour merge, and the resulting polyhedron becomes invalid.
651  if (connectingEdge->previous()->twin()->face() != opposingEdge->next()->twin()->face()) {
652  mergeNeighbours(connectingEdge->previous(), callback);
653  mergeNeighbours(opposingEdge->next(), callback);
654  } else {
655  mergeNeighbours(connectingEdge->next(), callback);
656  mergeNeighbours(opposingEdge->previous(), callback);
657  }
658 
659 
660  // Now we delete the destination of the connecting edge.
661  // First we have to change the origin of all edges originating
662  // at the destination to the origin of the connecting edge.
663  // We also have to delete the connecting edge and its twin from the incident faces.
664 
665  destination->setLeaving(connectingEdge->next());
666 
667  HalfEdge* firstEdge = destination->leaving();
668  HalfEdge* curEdge = firstEdge;
669  do {
670  HalfEdge* next = curEdge->nextIncident();
671  curEdge->setOrigin(origin);
672  curEdge = next;
673  } while (curEdge != firstEdge);
674 
675  origin->setLeaving(connectingEdge->previous()->twin());
676 
677  Face* leftFace = connectingEdge->face();
678  leftFace->removeFromBoundary(connectingEdge);
679 
680  Face* rightFace = opposingEdge->face();
681  rightFace->removeFromBoundary(opposingEdge);
682 
683  Edge* edge = connectingEdge->edge();
684  m_edges.remove(edge);
685  delete edge;
686 
687  delete connectingEdge;
688  delete opposingEdge;
689 
690  m_vertices.remove(destination);
691  delete destination;
692 }

741 template <typename T, typename FP, typename VP>
742 void Polyhedron<T,FP,VP>::mergeLeavingEdges(Vertex* vertex, Callback& callback) {
743  HalfEdge* curEdge = vertex->leaving();
744  do {
745  Vertex* destination = curEdge->destination();
746  HalfEdge* colinearEdge = destination->findColinearEdge(curEdge);
747  if (colinearEdge != NULL) {
748  mergeNeighboursOfColinearEdges(curEdge, colinearEdge, callback);

771 // Merges the neighbours of the given successive colinear edges with their coplanar neighbours.
772 template <typename T, typename FP, typename VP>
773 void Polyhedron<T,FP,VP>::mergeNeighboursOfColinearEdges(HalfEdge* edge1, HalfEdge* edge2, Callback& callback) {
774  assert(edge1->destination() == edge2->origin());
775 
776  if (edge1->face()->vertexCount() == 3 && edge1->next() == edge2) { // the left side will become a degenerate triangle now
777  mergeNeighbours(edge1->previous(), callback);
778  } else {
779  while (edge1->next()->face() != edge2->face()) // the face might already have been merged previously
780  mergeNeighbours(edge1->next(), callback);
781  }
782 
783  if (edge1->twin()->face()->vertexCount() == 3 && edge1->twin()->previous() == edge2->twin()) {
784  mergeNeighbours(edge1->twin()->next(), callback);
785  } else {
786  while (edge1->twin()->face() != edge2->twin()->face())
787  mergeNeighbours(edge1->twin()->previous(), callback);
788  }
789 }

791 // Merges the given successive colinear edges. As a result, the origin of the second
792 // given edge, the second given edge itself and its twin are removed. At the same time,
793 // the first given edge replaces the second given edge, and the first given edge's twin
794 // replaces the twin of the second given edge.
795 // Assumes that the incident faces have already been merged.
796 template <typename T, typename FP, typename VP>
797 typename Polyhedron<T,FP,VP>::Edge* Polyhedron<T,FP,VP>::mergeColinearEdges(HalfEdge* edge1, HalfEdge* edge2) {
798  assert(edge1->destination() == edge2->origin());
799  assert(edge1->face() == edge2->face());
800  assert(edge1->twin()->face() == edge2->twin()->face());
801 
802  Vertex* vertex = edge2->origin();
803  Vertex* succ = edge2->destination();
804 
805  edge1->twin()->setOrigin(succ);
806  succ->setLeaving(edge1->twin());
807 
808  HalfEdge* toRemove1 = edge2;
809  HalfEdge* toRemove2 = toRemove1->twin();
810 
811  Face* left = toRemove1->face();
812  left->removeFromBoundary(toRemove1);
813 
814  Face* right = toRemove2->face();
815  right->removeFromBoundary(toRemove2);
816 
817  Edge* edge = toRemove1->edge();
818  m_edges.remove(edge);
819  delete edge;
820 
821  delete toRemove1;
822  delete toRemove2;
823 
824  m_vertices.remove(vertex);
825  delete vertex;
826 
827  return edge1->edge();
828 }

830 // Merges all faces incident to the given vertex with their coplanar neighbours.
831 // Returns the given vertex or NULL if the given vertex was deleted.
832 template <typename T, typename FP, typename VP>
833 typename Polyhedron<T,FP,VP>::Face* Polyhedron<T,FP,VP>::mergeIncidentFaces(Vertex* vertex, Callback& callback) {
834  size_t incidentFaceCount = 0;
835  bool allCoplanar = true;
836 
837  // First determine whether all incident faces have become coplanar. While doing this,
838  // also remember an edge such that its incident faces are not coplanar.
839  HalfEdge* firstEdge = vertex->leaving();
840  HalfEdge* curEdge = firstEdge;
841  do {
842  Face* face = curEdge->face();
843  HalfEdge* innerBorder = curEdge->previous();
844  Face* innerNeighbour = innerBorder->twin()->face();
845 
846  if (!face->coplanar(innerNeighbour)) {
847  allCoplanar = false;
848  firstEdge = innerBorder->twin();
849  }
850  curEdge = innerBorder->twin();
851  ++incidentFaceCount;
852  } while (allCoplanar && curEdge != firstEdge);
853 
854  Face* remainingFace = NULL;
855 
856  if (allCoplanar) {
857  HalfEdgeList boundary;
858 
859  // Now we iterate using the incident face count because we can't rely
860  // on the curEdge's twin while we're deleting the edges we encounter.
861  curEdge = firstEdge;
862  remainingFace = curEdge->face();
863 
864  for (size_t i = 0; i < incidentFaceCount; ++i) {
865  Face* face = curEdge->face();
866  Edge* edge = curEdge->edge();
867 
868  HalfEdge* twin = curEdge->twin();
869  HalfEdge* outerBorder = curEdge->next();
870 
871  // Don't increment past the last edge because its twin will already have been deleted.
872  if (i < incidentFaceCount - 1)
873  curEdge = curEdge->nextIncident();
874 
875  twin->origin()->setLeaving(outerBorder);
876 
877  face->removeFromBoundary(outerBorder);
878  boundary.append(outerBorder, 1);
879 
880  if (i > 0) { // We want to keep the original face around
881  callback.facesWillBeMerged(remainingFace, face);
882  m_faces.remove(face);
883  delete face;
884  }
885 
886  m_edges.remove(edge);
887  delete edge;
888  }
889 
890  // also deletes the old boundary once the local boundary variable falls out of scope
891  remainingFace->replaceEntireBoundary(boundary);
892 
893  m_vertices.remove(vertex);
894  delete vertex;
895  } else {
896  assert(!firstEdge->face()->coplanar(firstEdge->twin()->face()));
897 
898  // We have chosen the first edge such that its incident faces are not coplanar.
899  // Now we start search for patches of coplanar faces and merge them.
900 
901  // Due to how we have chosen the first edge, we know that the first two
902  // incident faces are not coplanar and thus won't be merged. This is important
903  // because otherwise, the loop would immediately terminate after one iteration
904  // since curEdge is not incremented when two inner neighbours are merged.
905  curEdge = firstEdge;
906  bool didIncrement = false;
907  do {
908  Face* face = curEdge->face();
909 
910  HalfEdge* outerBorder = curEdge->next();
911  Face* outerNeighbour = outerBorder->twin()->face();
912 
913  HalfEdge* innerBorder = curEdge->previous();
914  Face* innerNeighbour = innerBorder->twin()->face();
915 
916  if (face->coplanar(innerNeighbour)) {
917  // Ensure that we don't remove the first edge, otherwise we'll loop endlessly.
918  mergeNeighbours(innerBorder, callback);
919  } else if (face->coplanar(outerNeighbour)) {
920  mergeNeighbours(outerBorder, callback);
921  curEdge = curEdge->nextIncident();
922  didIncrement = true;
923  } else {
924  curEdge = curEdge->nextIncident();
925  didIncrement = true;
926  }
927  } while (curEdge != firstEdge || !didIncrement);
928  }
929 
930  return remainingFace;
931 }

933 // Merges the face incident to the given edge (called border) with its neighbour, i.e.
934 // the face incident to the given face's twin. The face incident to the border is deleted
935 // while the neighbour consumes the boundary of the incident face.
936 // Also handles the case where the border is longer than just one edge.
937 template <typename T, typename FP, typename VP>
938 void Polyhedron<T,FP,VP>::mergeNeighbours(HalfEdge* borderFirst, Callback& callback) {

940  Face* neighbour = borderFirst->twin()->face();

943  borderFirst->previous()->twin()->face() == neighbour) {
944  borderFirst = borderFirst->previous();
945  }

948  HalfEdge* borderLast = borderFirst;

951  borderLast->next()->twin()->face() == neighbour) {
952  borderLast = borderLast->next();
953  }
954

956 
957  borderFirst->origin()->setLeaving(twinLast->next());
958  twinFirst->origin()->setLeaving(borderLast->next());
959

961  HalfEdge* remainingLast = borderFirst->previous();
962

965

967

972  HalfEdge* twin = cur->twin();
973  Vertex* origin = cur->origin();
974

976  delete edge;
977

983  delete origin;
984  }

987  } while (cur != borderFirst);
988

990  m_faces.remove(face);
