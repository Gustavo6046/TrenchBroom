176 template <typename T, typename FP, typename VP>
177 typename Polyhedron<T,FP,VP>::Edge* Polyhedron<T,FP,VP>::Edge::split(const Plane<T,3>& plane) {
178  // Do exactly what QBSP is doing:
179  const T startDist = plane.pointDistance(firstVertex()->position());
180  const T endDist = plane.pointDistance(secondVertex()->position());
181 
182  assert(startDist != endDist);
183  const T dot = startDist / (startDist - endDist);
184 
185  const V& startPos = firstVertex()->position();
186  const V& endPos = secondVertex()->position();
187  V position;
188  for (size_t i = 0; i < 3; ++i) {
189  if (plane.normal[i] == 1.0)
190  position[i] = plane.distance;
191  else if (plane.normal[i] == -1.0)
192  position[i] = -plane.distance;
193  else
194  position[i] = startPos[i] + dot * (endPos[i] - startPos[i]);
195  }
196 
197  return insertVertex(position);
198 }

206 typename Polyhedron<T,FP,VP>::Edge* Polyhedron<T,FP,VP>::Edge::insertVertex(const V& position) {
207  Vertex* newVertex = new Vertex(position);
208  HalfEdge* newFirstEdge = new HalfEdge(newVertex);
209  HalfEdge* oldFirstEdge = firstEdge();
210  HalfEdge* newSecondEdge = new HalfEdge(newVertex);
211  HalfEdge* oldSecondEdge = secondEdge();
212 
213  newFirstEdge->setFace(firstFace());
214  newSecondEdge->setFace(secondFace());
215 
216  firstFace()->insertIntoBoundaryAfter(oldFirstEdge, newFirstEdge);
217  secondFace()->insertIntoBoundaryAfter(oldSecondEdge, newSecondEdge);
218

222
