136 template <typename T, typename FP, typename VP>
137 typename Polyhedron<T,FP,VP>::V Polyhedron<T,FP,VP>::Face::normal() const {
138  const HalfEdge* first = m_boundary.front();
139  const HalfEdge* current = first;
140  V cross;
141  do {
142  const V& p1 = current->origin()->position();
143  const V& p2 = current->next()->origin()->position();
144  const V& p3 = current->next()->next()->origin()->position();
145  cross = crossed(p2 - p1, p3 - p1);
146  if (!cross.null())
147  return cross.normalized();
148  current = current->next();
149  } while (first != current);
150  return cross;
151 }

158 template <typename T, typename FP, typename VP>
159 T Polyhedron<T,FP,VP>::Face::intersectWithRay(const Ray<T,3>& ray, const Math::Side side) const {
160  const Plane<T,3> plane(origin(), normal());
161  const T dot = plane.normal.dot(ray.direction);
162  if (Math::zero(dot))
163  return Math::nan<T>();
164  if (side != Math::Side_Both) {
165  if (side == Math::Side_Front) {
166  if (dot > 0.0)
167  return Math::nan<T>();
168  } else if (dot < 0.0) { // and side == Math::Side_Back
169  return Math::nan<T>();
170  }
171  }
172 
173  return intersectPolygonWithRay(ray, plane, m_boundary.begin(), m_boundary.end(), GetVertexPosition());
174 }
