116 void Polyhedron<T,FP,VP>::addPoint(const V& position, Callback& callback) {
117  assert(checkInvariant());

119  case 0:
120  addFirstPoint(position);
121  m_bounds.min = m_bounds.max = position;
122  break;
123  case 1:

127  case 2:

219 //Adds the given point to a polygon. The result is either a differen polygon if the
220 // given point is coplanar to the already existing polygon, or a polyhedron if the
221 // given point is not coplanar.
222 template <typename T, typename FP, typename VP>
223 bool Polyhedron<T,FP,VP>::addFurtherPointToPolygon(const V& position, Callback& callback) {
224  Face* face = *m_faces.begin();
225  const Math::PointStatus::Type status = face->pointStatus(position);
226  switch (status) {
227  case Math::PointStatus::PSInside:
228  addPointToPolygon(position, callback);
229  return true;
230  case Math::PointStatus::PSAbove:
231  face->flip();
232  case Math::PointStatus::PSBelow:
233  return makePolyhedron(position, callback);
234  }
235  // will never be reached
236  return true;

252 // Creates a new polygon from the given set of coplanar points. Assumes that
253 // this polyhedron is empty and that the given point list contains at least three
254 // non-colinear points.
255 template <typename T, typename FP, typename VP>
256 void Polyhedron<T,FP,VP>::makePolygon(const typename V::List& positions, Callback& callback) {
257  assert(empty());
258  assert(positions.size() > 2);
259 
260  HalfEdgeList boundary;
261  for (size_t i = 0; i < positions.size(); ++i) {
262  const V& p = positions[i];
263  Vertex* v = new Vertex(p);
264  HalfEdge* h = new HalfEdge(v);
265  Edge* e = new Edge(h);
266 
267  m_vertices.append(v, 1);
268  boundary.append(h, 1);
269  m_edges.append(e, 1);
270  }
271 
272  Face* f = new Face(boundary);
273  callback.faceWasCreated(f);
274  m_faces.append(f, 1);
275 }

277 // Converts a coplanar polyhedron into a non-coplanar one by adding the given
278 // point, which is assumed to be non-coplanar to the points in this polyhedron.
279 template <typename T, typename FP, typename VP>
280 bool Polyhedron<T,FP,VP>::makePolyhedron(const V& position, Callback& callback) {
281  assert(polygon());
282 
283  Seam seam;
284  Face* face = m_faces.front();
285  const HalfEdgeList& boundary = face->boundary();
286 
287  HalfEdge* first = boundary.front();
288  HalfEdge* current = first;
289  do {
290  seam.push_back(current->edge());
291  current = current->previous(); // The seam must be CCW, so we have to iterate in reverse order in this case.
292  } while (current != first);
293 
294  return addPointToPolyhedron(position, seam, callback);
295 }

308 // Adds the given point to this polyhedron by weaving a cap over the given seam.
309 // Assumes that this polyhedron has been split by the given seam.
310 template <typename T, typename FP, typename VP>
311 bool Polyhedron<T,FP,VP>::addPointToPolyhedron(const V& position, const Seam& seam, Callback& callback) {
312  assert(!seam.empty());
313 
314  Face* remainingFace = m_faces.front();
315  typename V::List vertices;
316  vertices.reserve(remainingFace->vertexCount());
317  remainingFace->getVertexPositions(std::back_inserter(vertices));
318 
319  Vertex* newVertex = weaveCap(seam, position, callback);
320  cleanupAfterVertexMove(newVertex, callback);
321  if (faceCount() < 4) {
322  // If this polyhedron was a polygon, and the added point was too close to it, the cleanup
323  // may merge some of the newly added faces, and the result is an invalid polyhedron.
324  // We recreate the original polygon.
325 
326  clear();
327  addPoints(vertices.begin(), vertices.end(), callback);
328 
329  return false;
330  }
331  return true;
332 }

334 template <typename T, typename FP, typename VP>
335 typename Polyhedron<T,FP,VP>::Seam Polyhedron<T,FP,VP>::createSeam(const SplittingCriterion& criterion) {
336  Seam seam;
337 
338  Edge* first = criterion.findFirstSplittingEdge(m_edges);
339  if (first != NULL) {
340  Edge* current = first;
341  do {
342  assert(current != NULL);
343  seam.push_back(current);
344  current = criterion.findNextSplittingEdge(current);
345  } while (current != first);
346  }
347 
348  // The resulting seam contains the edges where one face satisfies the given criterion while the other does not.
349  // The edges are in counter clockwise order and consecutive, and they form a loop. They are oriented such that
350  // the first face matches the criterion and the second face does not.
351  return seam;
352 }

354 // Splits this polyhedron along the given seam and removes all faces, edges and vertices which are "above" the seam.
355 template <typename T, typename FP, typename VP>
356 void Polyhedron<T,FP,VP>::split(const Seam& seam, Callback& callback) {
357  assert(seam.size() >= 3);
358 
359  // First, unset the second half edge of every seam edge.
360  // Thereby remember the second half edge of the first seam edge.
361  HalfEdge* first = seam.front()->secondEdge();
362  typename Seam::const_iterator it, end;
363  for (it = seam.begin(), end = seam.end(); it != end; ++it) {
364  Edge* edge = *it;
365  edge->setFirstAsLeaving();
366  edge->unsetSecondEdge();
367  }
368 
369  // Now we must delete all the faces, edges, and vertices which are above the seam.
370  // The first half edge we remembered above is our entry point into that portion of the polyhedron.
371  // We must remember which faces we have already visited to stop the recursion.
372  FaceSet faceSet;
373  VertexList verticesToDelete;
374  deleteFaces(first, faceSet, verticesToDelete, callback);
375 }

377 template <typename T, typename FP, typename VP>
378 void Polyhedron<T,FP,VP>::deleteFaces(HalfEdge* first, FaceSet& visitedFaces, VertexList& verticesToDelete, Callback& callback) {
379  Face* face = first->face();
380  if (!visitedFaces.insert(face).second)
381  return;
382 
383  HalfEdge* current = first;
384  do {
385  Edge* edge = current->edge();
386  if (edge != NULL) {
387  if (edge->fullySpecified())
388  deleteFaces(edge->twin(current), visitedFaces, verticesToDelete, callback);
389 
390  if (edge->fullySpecified()) {
391  edge->makeSecondEdge(current);
392  edge->unsetSecondEdge();
393  } else {
394  current->setEdge(NULL);
395  m_edges.remove(edge);
396  delete edge;
397  }
398  }
399  Vertex* origin = current->origin();
400  if (origin->leaving() == current) {
401  m_vertices.remove(origin);
402  verticesToDelete.append(origin, 1);
403  }
404  current = current->next();
405  } while (current != first);
406 
407  callback.faceWillBeDeleted(face);
408  m_faces.remove(face);
409  delete face;
410 }

414 template <typename T, typename FP, typename VP>
415 typename Polyhedron<T,FP,VP>::Face* Polyhedron<T,FP,VP>::weaveCap(const Seam& seam, Callback& callback) {
416  assert(seam.size() >= 3);
417 
418  HalfEdgeList boundary;
419  typename Seam::const_iterator it, end;
420  for (it = seam.begin(), end = seam.end(); it != end; ++it) {
421  Edge* currentEdge = *it;
422  assert(!currentEdge->fullySpecified());

424  Vertex* origin = currentEdge->secondVertex();

429 
430  Face* face = new Face(boundary);
431  callback.faceWasCreated(face);
432  m_faces.append(face, 1);
433  return face;

436 // Weaves a new cap onto the given seam edges. The new cap will form a triangle fan (actually a cone) with a new vertex
437 // at the location of the given point being shared by all the newly created triangles.
438 template <typename T, typename FP, typename VP>
439 typename Polyhedron<T,FP,VP>::Vertex* Polyhedron<T,FP,VP>::weaveCap(const Seam& seam, const V& position, Callback& callback) {
440  assert(seam.size() >= 3);
441 
442  Vertex* top = new Vertex(position);
443 
444  HalfEdge* first = NULL;
445  HalfEdge* last = NULL;
446 
447  typename Seam::const_iterator it, end;
448  for (it = seam.begin(), end = seam.end(); it != end; ++it) {
449  Edge* edge = *it;
450  assert(!edge->fullySpecified());
451 
452  Vertex* v1 = edge->secondVertex();
453  Vertex* v2 = edge->firstVertex();
454 
455  HalfEdge* h1 = new HalfEdge(top);
456  HalfEdge* h2 = new HalfEdge(v1);
457  HalfEdge* h3 = new HalfEdge(v2);
458 
459  m_faces.append(createCapTriangle(h1, h2, h3, callback), 1);
460 
461  if (last != NULL)
462  m_edges.append(new Edge(h1, last), 1);
463  edge->setSecondEdge(h2);
464 
465  if (first == NULL)
466  first = h1;
467  last = h3;
468  }
469 
470  m_edges.append(new Edge(first, last), 1);
471  m_vertices.append(top, 1);
472 
473  return top;
474 }

488 template <typename T, typename FP, typename VP>
489 class Polyhedron<T,FP,VP>::SplittingCriterion {
490 private:
491  typedef enum {
492  MatchResult_First,
493  MatchResult_Second,
494  MatchResult_Both,
495  MatchResult_Neither
496  } MatchResult;
497 public:
498  virtual ~SplittingCriterion() {}
499 public:
500  Edge* findFirstSplittingEdge(EdgeList& edges) const {
501  typename EdgeList::iterator it, end;
502  for (it = edges.begin(), end = edges.end(); it != end; ++it) {
503  Edge* edge = *it;
504  const MatchResult result = matches(edge);
505  switch (result) {
506  case MatchResult_Second:
507  edge->flip();
508  case MatchResult_First:
509  return edge;
510  case MatchResult_Both:
511  case MatchResult_Neither:
512  break;
513  switchDefault()
514  }
515  }
516  return NULL;
517  }
518 
519  // finds the next seam edge in counter clockwise orientation
520  Edge* findNextSplittingEdge(Edge* last) const {
521  assert(last != NULL);
522 
523  HalfEdge* halfEdge = last->firstEdge()->previous();
524  Edge* next = halfEdge->edge();
525 
526  MatchResult result = matches(next);
527  while (result != MatchResult_First && result != MatchResult_Second && next != last) {
528  halfEdge = halfEdge->twin()->previous();
529  next = halfEdge->edge();
530  result = matches(next);
531  }
532 
533  if (result != MatchResult_First && result != MatchResult_Second)
534  return NULL;
535 
536  if (result == MatchResult_Second)
537  next->flip();
538  return next;
539  }
540 private:
541  MatchResult matches(const Edge* edge) const {
542  const bool firstResult = matches(edge->firstFace());
543  const bool secondResult = matches(edge->secondFace());
544  if (firstResult) {
545  if (secondResult)
546  return MatchResult_Both;
547  return MatchResult_First;
548  }
549  if (secondResult)
550  return MatchResult_Second;
551  return MatchResult_Neither;
552  }
553 public:
554  bool matches(const Face* face) const {
555  return doMatches(face);
556  }
557 private:
558  virtual bool doMatches(const Face* face) const = 0;
559 };
