42 template <typename T, typename FP, typename VP>
43 typename Polyhedron<T,FP,VP>::ClipResult Polyhedron<T,FP,VP>::clip(const Plane<T,3>& plane, Callback& callback) {
44  const ClipResult vertexResult = checkIntersects(plane);
45  if (!vertexResult.success())
46  return vertexResult;
47 
48  // Now we know that the brush will be split.
49  // The basic idea is now to split all faces which are intersected by the given plane so that the polyhedron
50  // can be separated into two halves such that no face has vertices on opposite sides of the plane.
51  const Seam seam = intersectWithPlane(plane, callback);
52 
53  // We construct a seam along those edges which are completely inside the plane and delete the half of the
54  // polyhedron that is above the plane. The remaining half is an open polyhedron (one face is missing) which
55  // is below the plane.
56  split(seam, callback);
57 
58  // We seal the polyhedron by creating a new face.
59  weaveCap(seam, callback);
60  updateBounds();
61 
62  assert(checkInvariant());
63  return ClipResult(ClipResult::Type_ClipSuccess);
64 }

66 template <typename T, typename FP, typename VP>
67 typename Polyhedron<T,FP,VP>::ClipResult Polyhedron<T,FP,VP>::checkIntersects(const Plane<T,3>& plane) const {
68  size_t above = 0;
69  size_t below = 0;
70  size_t inside = 0;
71 
72  typename VertexList::const_iterator it, end;
73  for (it = m_vertices.begin(), end = m_vertices.end(); it != end; ++it) {
74  const Vertex* vertex = *it;
75  const Math::PointStatus::Type status = plane.pointStatus(vertex->position());
76  switch (status) {
77  case Math::PointStatus::PSAbove:
78  ++above;
79  break;
80  case Math::PointStatus::PSBelow:
81  ++below;
82  break;
83  case Math::PointStatus::PSInside:
84  ++inside;
85  break;
86  switchDefault()
87  }
88  }
89 
90  assert(above + below + inside == m_vertices.size());
91  if (below + inside == m_vertices.size())
92  return ClipResult(ClipResult::Type_ClipUnchanged);
93  if (above + inside == m_vertices.size())
94  return ClipResult(ClipResult::Type_ClipEmpty);
95  return ClipResult(ClipResult::Type_ClipSuccess);
96 }

98 template <typename T, typename FP, typename VP>
99 typename Polyhedron<T,FP,VP>::Seam Polyhedron<T,FP,VP>::intersectWithPlane(const Plane<T,3>& plane, Callback& callback) {
100  Seam seam;
101 
102  // First, we find a half edge that is intersected by the given plane.
103  HalfEdge* initialEdge = findInitialIntersectingEdge(plane);
104  assert(initialEdge != NULL);
105 
106  // Now we split the face to which this initial half edge belongs. The call returns the newly inserted edge
107  // that connectes the (possibly newly inserted) vertices which are now within the plane.
108  HalfEdge* currentEdge = intersectWithPlane(initialEdge, plane, callback);
109 
110  // The destination of that edge is the first vertex which we encountered (or inserted) which is inside the plane.
111  // This is where our algorithm must stop. When we encounter that vertex again, we have completed the intersection
112  // and the polyhedron can now be split in two by the given plane.
113  Vertex* stopVertex = currentEdge->destination();
114  do {
115  // First we find the next face that is either split by the plane or which has an edge completely in the plane.
116  currentEdge = findNextIntersectingEdge(currentEdge, plane);
117  assert(currentEdge != NULL);
118 
119  // Now we split that face. Again, the returned edge connects the two (possibly inserted) vertices of that
120  // face which are now inside the plane.
121  currentEdge = intersectWithPlane(currentEdge, plane, callback);
122 
123  // Build a seam while intersecting the polyhedron by remembering the edges we just inserted. To ensure that
124  // the seam edges are correctly oriented, we check that the current edge is the second edge, as the current
125  // edge belongs to the faces that we are going to clip away.
126  Edge* seamEdge = currentEdge->edge();
127  seamEdge->makeSecondEdge(currentEdge);
128  seam.push_back(seamEdge);
129 
130  } while (currentEdge->destination() != stopVertex);
131 
132  return seam;
133 }

135 template <typename T, typename FP, typename VP>
136 typename Polyhedron<T,FP,VP>::HalfEdge* Polyhedron<T,FP,VP>::findInitialIntersectingEdge(const Plane<T,3>& plane) const {
137  Edge* firstEdge = m_edges.front();
138  Edge* currentEdge = firstEdge;
139  do {
140  HalfEdge* halfEdge = currentEdge->firstEdge();
141  const Math::PointStatus::Type os = plane.pointStatus(halfEdge->origin()->position());
142  const Math::PointStatus::Type ds = plane.pointStatus(halfEdge->destination()->position());
143  if (os == Math::PointStatus::PSInside && ds == Math::PointStatus::PSInside) {
144  // If both ends of the edge are inside the plane, we must ensure that we return the correct
145  // half edge, which is either the current one or its twin. Since the returned half edge is supposed
146  // to be clipped away, we must examine the destination of its successor. If that is below the plane,
147  // we return the twin, otherwise we return the half edge.
148  const Math::PointStatus::Type ss = plane.pointStatus(halfEdge->next()->destination()->position());
149  assert(ss != Math::PointStatus::PSInside);
150  if (ss == Math::PointStatus::PSBelow)
151  return halfEdge->twin();
152  return halfEdge;
153  }
154 
155  if ((os == Math::PointStatus::PSInside && ds == Math::PointStatus::PSAbove) ||
156  (os == Math::PointStatus::PSBelow && ds == Math::PointStatus::PSAbove))
157  return halfEdge->twin();
158 
159  if (
160  (os == Math::PointStatus::PSAbove && ds == Math::PointStatus::PSInside) ||
161  (os == Math::PointStatus::PSAbove && ds == Math::PointStatus::PSBelow))
162  return halfEdge;
163  currentEdge = currentEdge->next();
164  } while (currentEdge != firstEdge);
165  return NULL;
166 }

168 template <typename T, typename FP, typename VP>
169 typename Polyhedron<T,FP,VP>::HalfEdge* Polyhedron<T,FP,VP>::intersectWithPlane(HalfEdge* firstBoundaryEdge, const Plane<T,3>& plane, Callback& callback) {
170 
171  // Starting at the given edge, we search the boundary of the current face until we find an edge that is either split in two by the given plane
172  // or where its origin is inside it. In the first case, we split the found edge by inserting a vertex at the position where
173  // the plane intersects the edge. We remember the half edge starting at the newly inserted vertex as the seam origin or destination, depending
174  // on whether it's the first or second such edge we have found. In the second case (the edge's origin is inside the plane), we just store the
175  // half edge as either the seam origin or destination.
176  // In the end, we have two vertices, identified by half edges belonging to the currently treated face, which lie inside the plane. If these two
177  // vertices aren't already connected by an edge, we split the current face in two by inserting a new edge from the origin to the destination vertex.
178  // Finally we must decide where to continue our search, that is, we find a face that is incident to the destination vertex such that it is split
179  // by the given plane. We return the half edge of that face's boundary which starts in the destination vertex so that the search can continue there.
180 
181  HalfEdge* seamOrigin = NULL;
182  HalfEdge* seamDestination = NULL;
183 
184  HalfEdge* currentBoundaryEdge = firstBoundaryEdge;
185  do {
186  const Math::PointStatus::Type os = plane.pointStatus(currentBoundaryEdge->origin()->position());
187  const Math::PointStatus::Type ds = plane.pointStatus(currentBoundaryEdge->destination()->position());
188 
189  if (os == Math::PointStatus::PSInside) {
190  if (seamOrigin == NULL)
191  seamOrigin = currentBoundaryEdge;
192  else
193  seamDestination = currentBoundaryEdge;
194  currentBoundaryEdge = currentBoundaryEdge->next();
195  } else if ((os == Math::PointStatus::PSBelow && ds == Math::PointStatus::PSAbove) ||
196  (os == Math::PointStatus::PSAbove && ds == Math::PointStatus::PSBelow)) {
197  // We have to split the edge and insert a new vertex, which will become the origin or destination of the new seam edge.
198  Edge* currentEdge = currentBoundaryEdge->edge();
199  Edge* newEdge = currentEdge->split(plane);
200  m_edges.append(newEdge, 1);
201 
202  currentBoundaryEdge = currentBoundaryEdge->next();
203  Vertex* newVertex = currentBoundaryEdge->origin();
204  m_vertices.append(newVertex, 1);
205 
206  // The newly inserted vertex will be reexamined in the next loop iteration as it is now contained within the plane.
207  } else {
208  currentBoundaryEdge = currentBoundaryEdge->next();
209  }
210  } while (seamDestination == NULL && currentBoundaryEdge != firstBoundaryEdge);
211  assert(seamOrigin != NULL);
212 
213  // The plane only touches one vertex of the face.
214  if (seamDestination == NULL)
215  return seamOrigin->previous();
216 
217  if (seamDestination->next() == seamOrigin) {
218  std::swap(seamOrigin, seamDestination);
219  } else if (seamOrigin->next() != seamDestination) {
220  // If the origin and the destination are not already connected by an edge, we must split the current face and insert an edge
221  // between them.
222  // The newly created faces are supposed to be above the given plane, so we have to consider whether the destination of the
223  // seam origin edge is above or below the plane.
224  const Math::PointStatus::Type os = plane.pointStatus(seamOrigin->destination()->position());
225  assert(os != Math::PointStatus::PSInside);
226  if (os == Math::PointStatus::PSBelow)
227  intersectWithPlane(seamOrigin, seamDestination, callback);
228  else
229  intersectWithPlane(seamDestination, seamOrigin, callback);
230  }
231 
232  return seamDestination->previous();
233 }

236 void Polyhedron<T,FP,VP>::intersectWithPlane(HalfEdge* oldBoundaryFirst, HalfEdge* newBoundaryFirst, Callback& callback) {
237  HalfEdge* newBoundaryLast = oldBoundaryFirst->previous();

239  HalfEdge* oldBoundarySplitter = new HalfEdge(newBoundaryFirst->origin());

241

243  newBoundarySplitter->setFace(oldFace);
244 
245  oldFace->insertIntoBoundaryAfter(newBoundaryLast, newBoundarySplitter);
246  const size_t newBoundaryCount = oldFace->replaceBoundary(newBoundaryFirst, newBoundarySplitter, oldBoundarySplitter);
247

260 template <typename T, typename FP, typename VP>
261 typename Polyhedron<T,FP,VP>::HalfEdge* Polyhedron<T,FP,VP>::findNextIntersectingEdge(HalfEdge* searchFrom, const Plane<T,3>& plane) const {
262  HalfEdge* currentEdge = searchFrom->next()->twin()->next();
263  const HalfEdge* stopEdge = searchFrom->next();
264  do {
265  // Select two vertices that form a triangle (of an adjacent face) together with seamDestination's origin vertex.
266  // If either of the two vertices is inside the plane or if they lie on different sides of it, then we have found
267  // the next face to handle.
268 
269  Vertex* cd = currentEdge->destination();
270  Vertex* po = currentEdge->previous()->origin();
271  const Math::PointStatus::Type cds = plane.pointStatus(cd->position());
272  const Math::PointStatus::Type pos = plane.pointStatus(po->position());
273 
274  if ((cds == Math::PointStatus::PSInside) ||
275  (cds == Math::PointStatus::PSBelow && pos == Math::PointStatus::PSAbove) ||
276  (cds == Math::PointStatus::PSAbove && pos == Math::PointStatus::PSBelow))
277  return currentEdge;
278 
279  currentEdge = currentEdge->twin()->next();
280  } while (currentEdge != stopEdge);
281  return NULL;
282 }
