65 void Color::rgbToHSB(const float r, const float g, const float b, float& h, float& s, float& br) {
66  assert(r >= 0.0f && r <= 1.0f);
67  assert(g >= 0.0f && g <= 1.0f);
68  assert(b >= 0.0f && b <= 1.0f);
69 
70  const float max = std::max(std::max(r, g), b);
71  const float min = std::min(std::min(r, g), b);
72  const float dist = max - min;
73 
74  br = max;
75  if (br != 0.0f)
76  s = dist / max;
77  else
78  s = 0.0f;
79 
80  if (s == 0.0f) {
81  h = 0.0f;
82  } else {
83  const float rc = (max - r) / dist;
84  const float gc = (max - g) / dist;
85  const float bc = (max - b) / dist;
86  if (r == max)
87  h = bc - gc;
88  else if (g == max)
89  h = 2.0f + rc - bc;
90  else
91  h = 4.0f + gc - rc;
92  h = h / 6.0f;
93  if (h < 0)
94  h = h + 1.0f;
95  }
