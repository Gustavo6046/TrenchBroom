28 template <typename T>
29 int handlePolygonEdgeIntersection(const Vec<T,3>& v0, const Vec<T,3>& v1) {
30  if ((Math::zero(v0.x()) && Math::zero(v0.y())) ||
31  (Math::zero(v1.x()) && Math::zero(v1.y()))) {
32  // the point is identical to a polygon vertex, cancel search
33  return -1;
34  }
35 
36  /*
37  * A polygon edge intersects with the positive X axis if the
38  * following conditions are met: The Y coordinates of its
39  * vertices must have different signs (we assign a negative sign
40  * to 0 here in order to count it as a negative number) and one
41  * of the following two conditions must be met: Either the X
42  * coordinates of the vertices are both positive or the X
43  * coordinates of the edge have different signs (again, we
44  * assign a negative sign to 0 here). In the latter case, we
45  * must calculate the point of intersection between the edge and
46  * the X axis and determine whether its X coordinate is positive
47  * or zero.
48  */
49 
50  // do the Y coordinates have different signs?
51  if (( Math::pos(v0.y()) && !Math::pos(v1.y())) ||
52  (!Math::pos(v0.y()) && Math::pos(v1.y()))) {
53  // Is segment entirely on the positive side of the X axis?
54  if (Math::pos(v0.x()) && Math::pos(v1.x())) {
55  return 1; // Edge intersects with the X axis.
56  }
57 
58  // If not, do the X coordinates have different signs?
59  if (( Math::pos(v0.x()) && !Math::pos(v1.x())) ||
60  (!Math::pos(v0.x()) && Math::pos(v1.x()))) {
61  // Calculate the point of intersection between the edge and the X axis.
62  const T x = -v0.y() * (v1.x() - v0.x()) / (v1.y() - v0.y()) + v0.x();
63  if (!Math::neg(x))
64  return 1; // Edge intersects with the X axis.
65  }
66  }
67 
68  return 0;
69 }

71 template <typename T, typename I, typename F>
72 T intersectPolygonWithRay(const Ray<T,3>& ray, const Plane<T,3> plane, I cur, I end, F getPosition) {
73  const T distance = plane.intersectWithRay(ray);
74  if (Math::isnan(distance))
75  return distance;
76 
77  const size_t axis = plane.normal.firstComponent();
78  const Vec<T,3> o = swizzle(ray.pointAtDistance(distance), axis);
79 
80  const Vec<T,3> f = swizzle(getPosition(*cur++), axis) - o; // The first vertex.
81  Vec<T,3> p = f; // The previous vertex.
82 
83  int d = 0;
84  while (cur != end) {
85  const Vec<T,3> c = swizzle(getPosition(*cur++), axis) - o; // The current vertex.
86  const int s = handlePolygonEdgeIntersection(p, c);
87  if (s == -1)
88  return distance;
89  d += s;
90  p = c;
91  }
92 
93  // Handle the edge from the last to the first vertex.
94  const int s = handlePolygonEdgeIntersection(p, f);
95  if (s == -1)
96  return distance;
97 
98  d += s;
99  if (d % 2 == 0)
100  return Math::nan<T>();
101  return distance;
102 }

104 /*
105  Returns > 0 if p3.xy() is to the left of the line through p1.xy() and p2.xy(),
106  < 0 if it is to the right of that line, or
107  = 0 if it is on the line.
108  */
109 template <typename T, size_t S>
110 int isLeft(const Vec<T,S>& p1, const Vec<T,S>& p2, const Vec<T,S>& p3) {
111  assert(S >= 2);
112  const T result = ( (p2.x() - p1.x()) * (p3.y() - p1.y())
113  -(p3.x() - p1.x()) * (p2.y() - p1.y()));
114  if (result < 0.0)
115  return -1;
116  if (result > 0.0)
117  return 1;
118  return 0;
119 }

122 template <typename T>
123 class ConvexHull2D {
124 private:
125  class LessThanByAngle {
126  private:
127  const Vec<T,3>& m_anchor;
128  public:
129  LessThanByAngle(const Vec<T,3>& anchor) : m_anchor(anchor) {}
130  public:
131  bool operator()(const Vec<T,3>& lhs, const Vec<T,3>& rhs) const {
132  const int side = isLeft(m_anchor, lhs, rhs);
133  if (side > 0)
134  return true;
135  if (side < 0)
136  return false;
137 
138  // the points are colinear, the one that is further from the anchor is considered less
139  const T dxl = Math::abs(lhs.x() - m_anchor.x());
140  const T dxr = Math::abs(rhs.x() - m_anchor.x());
141  if (dxl == dxr) {
142  const T dyl = Math::abs(lhs.y() - m_anchor.y());
143  const T dyr = Math::abs(rhs.y() - m_anchor.y());
144  return dyl > dyr;
145  }
146  return dxl > dxr;
147  }
148  };

206  void findAnchor() {
207  size_t anchor = 0;
208  for (size_t i = 1; i < m_points.size(); ++i) {
209  if ((m_points[i].y() < m_points[anchor].y()) ||
210  (m_points[i].y() == m_points[anchor].y() &&
211  m_points[i].x() > m_points[anchor].x()))
212  anchor = i;
213  }
214 
215  if (anchor > 0) {
216  using std::swap;
217  swap(m_points[0], m_points[anchor]);
218  }

221  void sortPoints() {
222  const Vec<T,3>& anchor = m_points[0];
223  std::sort(m_points.begin() + 1, m_points.end(), LessThanByAngle(anchor));
224 
225  // now remove the duplicates
226  typename Vec<T,3>::List::iterator i = m_points.begin() + 1;
227  while (i != m_points.end()) {
228  const Vec<T,3>& p1 = *(i++);
229  while (i != m_points.end()) {
230  const Vec<T,3>& p2 = *i;
231  if (isLeft(anchor, p1, p2) == 0)
232  i = m_points.erase(i);
233  else
234  break;
235  };
236  }
237  }

239  void buildHull() {
240  typename Vec<T,3>::List stack;
241  stack.reserve(m_points.size());
242  stack.push_back(m_points[0]);
243  stack.push_back(m_points[1]);
244 
245  for (size_t i = 2; i < m_points.size(); ++i) {
246  const Vec<T,3>& p = m_points[i];
247  popStalePoints(stack, p);
248  stack.push_back(p);
249  }
250 
251  using std::swap;
252  swap(m_points, stack);
253  assert(m_points.size() > 2);
254  }

256  void popStalePoints(typename Vec<T,3>::List& stack, const Vec<T,3>& p) {
257  if (stack.size() > 1) {
258  const Vec<T,3>& t1 = stack[stack.size() - 2];
259  const Vec<T,3>& t2 = stack[stack.size() - 1];
260  const int side = isLeft(t1, t2, p);
261  if (side < 0) {
262  stack.pop_back();
263  popStalePoints(stack, p);
264  }
265  }
266  }
