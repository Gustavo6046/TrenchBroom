47  Line<T,S> makeCanonical() const {
48  // choose the point such that its support vector is orthogonal to
49  // the direction of this line
50  const T d = point.dot(direction);
51  const Vec<T,S> newPoint(point - d * direction);
52 
53  // make sure the first nonzero component of the direction is positive
54  Vec<T,S> newDirection(direction);
55  for (size_t i = 0; i < S; ++i) {
56  if (direction[i] != 0.0) {
57  if (direction[i] < 0.0)
58  newDirection = -newDirection;
59  break;
60  }
61  }
62 
63  return Line<T,S>(newPoint, newDirection);
64  }

66  int compare(const Line<T,S>& other, const T epsilon = static_cast<T>(0.0)) const {
67  assert(isCanonical() && other.isCanonical());

70  if (pointCmp < 0)

122 template <typename TT>
123 const TT intersectLineWithTriangle(const Line<TT,3>& L, const Vec<TT,3>& V0, const Vec<TT,3>& V1, const Vec<TT,3>& V2) {
124  // see http://www.cs.virginia.edu/~gfx/Courses/2003/ImageSynthesis/papers/Acceleration/Fast%20MinimumStorage%20RayTriangle%20Intersection.pdf
125 
126  const Vec<TT,3>& O = L.point;
127  const Vec<TT,3>& D = L.direction;
128  const Vec<TT,3> E1 = V1 - V0;
129  const Vec<TT,3> E2 = V2 - V0;
130  const Vec<TT,3> P = crossed(D, E2);
131  const TT a = P.dot(E1);
132  if (Math::zero(a))
133  return Math::nan<TT>();
134 
135  const Vec<TT,3> T = O - V0;
136  const Vec<TT,3> Q = crossed(T, E1);
137 
138  const TT t = Q.dot(E2) / a;
139  const TT u = P.dot(T) / a;
140  if (Math::neg(u))
141  return Math::nan<TT>();
142 
143  const TT v = Q.dot(D) / a;
144  if (Math::neg(v))
145  return Math::nan<TT>();
146 
147  if (Math::gt(u+v, static_cast<TT>(1.0)))
148  return Math::nan<TT>();
149 
150  return t;
151 }
