233  template <typename T>
234  bool relEq(const T v1, const T v2, const T epsilon = Constants<T>::almostZero()) {
235  const T absA = abs(v1);
236  const T absB = abs(v2);
237  const T diff = abs(v1 - v2);
238 
239  if (v1 == v2) { // shortcut, handles infinities
240  return true;
241  } else if (v1 == 0.0 || v2 == 0.0 || diff < std::numeric_limits<T>::min()) {
242  // a or b is zero or both are extremely close to it
243  // relative error is less meaningful here
244  return diff < (epsilon * std::numeric_limits<T>::min());
245  } else { // use relative error
246  return diff / (absA + absB) < epsilon;
247  }
248  }
